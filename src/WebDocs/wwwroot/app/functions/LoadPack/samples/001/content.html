Configuration Example

// Drapo configuration example for defining packs:
drapo.Config.Packs
    .Add("testpack")
        .AddPath("testpack/*.html")
        .AddPath("testpack/*.css")
        .AddPath("testpack/*.js")
    .Add("uicomponents")
        .AddPath("components/**/*.html")
        .AddPath("styles/components/*.css")
        .ExcludePath("components/internal/*");
